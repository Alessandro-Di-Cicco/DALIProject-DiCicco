/* Utility procedures to sleep for a random period and to 
fetch a random element from a list */
pause:-
    random(1, 5, Len),
    sleep(Len).

random_element(List, Element):-
    length(List, Length),
    random(1, Length + 1, Index),
    nth1(Index, List, Element).
/* ------------------------------------------------------*/


requestI(Num):>
    Num > 0,
    retract(request(Num)),
    N is Num - 1,

    perform_request,

    pause,

    assert(request(N)).


requestI(0):>
    retract(request(0)),
    perform_request,

    pause,

    assert(refills(5)).


perform_request:-
    cities(CityList),
    materials(MaterialList),
    amount_bounds(Min, Max),

    random_element(CityList, City),
    random_element(MaterialList, Material),
    random(Min, Max, Amount),
    messageA(coordinator, send_message(deliver(City, Material, Amount, Me), Me)),
    format('Requested ~w ~w to ~w~n', [Amount, Material, City]).
